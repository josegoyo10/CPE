SELECT o.id_os, 
                  o.id_proyecto, 
                  o.id_local, 
                  o.clie_rut, 
                  o.os_fechacreacion, 
                  o.os_comentarios, 
                  o.usuario, 
                  o.USR_ID, 
                  o.USR_ORIGEN,
				  o.os_fechaboleta, 
				  o.os_numboleta, 
				  o.os_codbarras, 
				  o.os_terminal_pos, 
				  h.id_historial, 
				  h.id_os as id_os2, 
				  h.hist_usuario, 
				  h.hist_descripcion,
				  l.cod_local, 
				  l.nom_local, 
				  l.dir_local, 
				  l.ciu_local, 
				  osde.osde_tipoprod, 
				  osde.osde_subtipoprod, 
				  osde.osde_precio, 
				  osde.osde_cantidad,
                  osde.osde_descuento, 
                  osde.osde_preciocosto, 
                  osde.cod_sap, 
                  osde.osde_descripcion, 
                  osde.id_producto, 
                  osde.cod_barra, 
                  osde.cant_pickeada,
				 osde.osde_fecha_entrega 
		FROM os o JOIN historial h ON o.id_os = h.id_os
		JOIN locales l ON l.id_local = o.id_local
		JOIN os_detalle osde ON o.id_os = osde.id_os
		WHERE hist_descripcion LIKE '%Pagada%'
		and os_fechacreacion between '2015-08-17 00:00:00' and '2015-09-05 23:59:59'
